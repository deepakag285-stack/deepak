let lastSeenId = 0; // Sabse pehli baar koi id nahi dikhi

async function checkForUpdates() {
  try {
    const response = await fetch('/data.json');
    const data = await response.json();

    if (data.updates) {
      data.updates.forEach((update) => {
        // Sirf wahi update dikhao jo lastSeenId se badi ho
        if (update.id > lastSeenId) {
          showNotif(update.title, update.body, update.id);
          
          // Current ID ko update kar do taake next time ye dobara na aaye
          lastSeenId = update.id;
        }
      });
    }
  } catch (err) {
    console.log("Error:", err);
  }
}

let lastKnownBody = ""; // Pehle body kuch nahi thi

async function checkForUpdates() {
  try {
    const response = await fetch('/data.json');
    const data = await response.json();

    // Check karein kya nayi body purani body se alag hai?
    if (data.body !== lastKnownBody) {
      showNotif(data.title, data.body);
      
      // Ab nayi body ko save kar lo taake agli baar compare ho sake
      lastKnownBody = data.body;
    }
  } catch (err) {
    console.log("Error:", err);
  }
}

    function showNotif(title, body) {
      if (Notification.permission === 'granted') {
        navigator.serviceWorker.ready.then((reg) => {
          reg.showNotification(title, { body: body, icon: 'favicon.ico' });
        });
      }
    }

let messageCount = parseInt(localStorage.getItem('notifCount') || 0);

async function checkForUpdates() {
  try {
    const response = await fetch('/data.json');
    const data = await response.json();

    // Condition 1: Body change honi chahiye
    // Condition 2: Count 2 se kam honi chahiye
    if (data.body !== lastKnownBody && messageCount < 2) {
      
      showNotif(data.title, data.body);
      
      // Update values
      lastKnownBody = data.body;
      messageCount++; // Count ek badha do
      
      // Browser mein save kar do taake refresh hone par bhi yaad rahe
      localStorage.setItem('notifCount', messageCount);
      
      console.log("Message sent count: " + messageCount);
    } 
    else if (messageCount >= 2) {
      console.log("Max notification limit reached.");
    }
